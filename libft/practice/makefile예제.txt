BONUS_OBJS = $(BONUS_SRCS:.c=.o)
OBJS = $(SRCS:.c=.o)
NAME = libft.a

$(NAME) : $(OBJS)
    ar rc $(NAME) $(OBJS)

%.o : %.c
    $(CC) $(CFLAGS) -c $< -o $@

.PHONY: all
all : $(NAME)

.PHONY: clean
clean :
    rm -f $(OBJS) $(BONUS_OBJS)

.PHONY: fclean
fclean : clean
    rm -f $(NAME)

.PHONY: re
re : fclean all

.PHONY: bonus
bonus : $(OBJS) $(BONUS_OBJS)
    ar rc $(NAME) $^
